//===== DimensionsRO ==========================================
//= Phase 7 — Validation Kit (GM-only)
//===== Description: =========================================
//= Tools for validating Phase 3-6 balance changes:
//=
//=   1. DPS Dummy NPC — invulnerable target with HP counter that
//=      shows damage taken over a 30-second window. Use it to
//=      benchmark each class' burst + sustain DPS.
//=
//=   2. Test Char Outfitter — instantly equips the GM with a full
//=      "tier 1 expected gear" set (Scarlet weapon +9, Diabolus
//=      armor +7, Variant Shoes +7, Brisingamen x2) for the
//=      current job, plus job-appropriate consumables. Wraps the
//=      same loadout as dro_starter_kit.txt but free + unlimited.
//=
//= Both restricted to group_id >= 99 (GM full admin).
//============================================================

prontera,158,200,4	script	DRO Validation - DPS Dummy	4_BULLETIN_BOARD,{
	if (getgroupid() < 99) {
		mes "[DPS Dummy]";
		mes "GM only.";
		close;
	}
	mes "[DPS Dummy]";
	mes "I track damage taken over the next 30 seconds. Hit me with whatever skill rotation you want to test.";
	mes " ";
	mes "Current accumulated damage: ^228822" + .total_dmg + "^000000";
	mes "Window time elapsed: ^228822" + (.window_active ? (gettimetick(2) - .window_start) : 0) + "s^000000";
	mes " ";
	switch (select("Start new 30s window:Read current total:Reset:Cancel")) {
	case 1:
		.window_start = gettimetick(2);
		.window_active = 1;
		.total_dmg = 0;
		announce sd_name() + " started a DPS test window. 30s timer running.", bc_self;
		break;
	case 2:
		mes "Total damage so far: " + .total_dmg;
		mes "DPS = " + (.total_dmg / 30) + " /sec average";
		break;
	case 3:
		.total_dmg = 0;
		.window_active = 0;
		mes "Reset.";
		break;
	}
	close;

OnTouchDamage:
	if (.window_active && (gettimetick(2) - .window_start) <= 30) {
		.total_dmg += getarg(0);
	} else if (.window_active) {
		.window_active = 0;
		announce "DPS window closed. Total: " + .total_dmg + " (avg " + (.total_dmg / 30) + " /sec)", bc_self;
	}
	end;

OnInit:
	.total_dmg = 0;
	.window_active = 0;
	end;
}

prontera,160,200,4	script	DRO Validation - Outfitter	4_M_KAGE,{
	if (getgroupid() < 99) {
		mes "[Validation Outfitter]";
		mes "GM only.";
		close;
	}
	if (BaseLevel < 99) {
		mes "[Validation Outfitter]";
		mes "Set your level >= 99 first (@blvl).";
		close;
	}
	mes "[Validation Outfitter]";
	mes "Equips a free tier-1 testing kit for your current class.";
	mes "Same loadout as the Adventurer Starter Kit but free and repeatable.";
	next;
	if (select("Equip kit:Cancel") == 2) close;

	.@weapon = 0;
	switch (BaseJob) {
		case Job_Knight:         .@weapon = 21015; break;
		case Job_Crusader:       .@weapon = 1498;  break;
		case Job_Blacksmith:     .@weapon = 28106; break;
		case Job_Alchemist:      .@weapon = 16040; break;
		case Job_Monk:           .@weapon = 1839;  break;
		case Job_Priest:         .@weapon = 16040; break;
		case Job_Wizard:         .@weapon = 15313; break;
		case Job_Sage:           .@weapon = 15313; break;
		case Job_Hunter:         .@weapon = 18130; break;
		case Job_Bard:           .@weapon = 19550; break;
		case Job_Dancer:         .@weapon = 20844; break;
		case Job_Rogue:          .@weapon = 28705; break;
		case Job_Assassin:       .@weapon = 28007; break;
		case Job_Ninja:          .@weapon = 13327; break;
		case Job_Gunslinger:     .@weapon = 13127; break;
		case Job_Star_Gladiator: .@weapon = 1839;  break;
		case Job_Soul_Linker:    .@weapon = 15313; break;
		case Job_Summoner:       .@weapon = 16040; break;
		default:                 .@weapon = 16040; break;
	}
	getitem2 .@weapon, 1, 1, 9, 0, 0, 0, 0, 0;
	getitem2 28102, 1, 1, 7, 0, 0, 0, 0, 0;  // Diabolus Robe[1]
	getitem2 31925, 1, 1, 7, 0, 0, 0, 0, 0;  // Diabolus Manteau
	getitem2 29287, 1, 1, 7, 0, 0, 0, 0, 0;  // Variant Shoes
	getitem 2630, 2;                          // Brisingamen x2

	// Job-appropriate consumables
	getitem 12111, 50;  // Box of Drowsiness (50 white potions)
	getitem 547, 100;   // White Potion
	getitem 545, 100;   // Red Slim Potion

	mes "[Validation Outfitter]";
	mes "Tier-1 kit equipped. Test away.";
	close;
}
