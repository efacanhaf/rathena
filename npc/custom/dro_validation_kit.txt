//===== DimensionsRO ==========================================
//= Phase 7 — Validation Kit (GM-only)
//===== Description: =========================================
//= Tools for validating Phase 3-6 balance changes:
//=
//=   1. DPS Dummy NPC — invulnerable target with HP counter that
//=      shows damage taken over a 30-second window. Use it to
//=      benchmark each class' burst + sustain DPS.
//=
//=   2. Test Char Outfitter — instantly equips the GM with a full
//=      "tier 1 expected gear" set (Scarlet weapon +9, Diabolus
//=      armor +7, Variant Shoes +7, Brisingamen x2) for the
//=      current job, plus job-appropriate consumables. Wraps the
//=      same loadout as dro_starter_kit.txt but free + unlimited.
//=
//= Both restricted to group_id >= 99 (GM full admin).
//============================================================

prontera,158,177,4	script	DRO_DPS_DUMMY	4_BULLETIN_BOARD2,{
	if (getgroupid() < 99) {
		mes "[DPS Dummy]";
		mes "GM only.";
		close;
	}
	mes "[DPS Dummy]";
	mes "Last result:";
	mes "  Total dmg: ^228822" + .last_dmg + "^000000";
	mes "  Avg DPS:   ^228822" + (.last_dmg / 30) + "^000000";
	mes " ";
	mes "Starting a test will spawn a high-HP target near me at (159,177). You attack it. After 30s I read the HP delta.";
	switch (select("Start 30s test:Cancel")) {
	case 1:
		killmonster "prontera", "DRO_DPS_DUMMY::OnDummyDeath";
		monster "prontera", 165, 177, "DPS Target", 1002, 1, "DRO_DPS_DUMMY::OnDummyDeath";
		.dummy_gid = $@mobid[0];
		setunitdata .dummy_gid, UMOB_MAXHP, 2000000000;
		setunitdata .dummy_gid, UMOB_HP,    2000000000;
		setunitdata .dummy_gid, UMOB_DEF, 0;
		setunitdata .dummy_gid, UMOB_MDEF, 0;
		setunitdata .dummy_gid, UMOB_MODE, 0;          // no AI / no aggro / no movement (CANMOVE bit cleared)
		setunitdata .dummy_gid, UMOB_ATKMIN, 0;        // disarm
		setunitdata .dummy_gid, UMOB_ATKMAX, 0;
		setunitdata .dummy_gid, UMOB_ELEMENT, 0;       // Neutral
		setunitdata .dummy_gid, UMOB_ELEMENT_LEVEL, 1;
		setunitdata .dummy_gid, UMOB_SPEED, 1000;      // dampen even if engine ignores MODE
		.start_hp = 2000000000;
		stopnpctimer;
		initnpctimer;
		announce strcharinfo(0) + " started a DPS test. 30s — HIT THE TARGET.", bc_self;
		break;
	}
	close;

OnTimer30000:
	stopnpctimer;
	if (!.dummy_gid) end;
	getunitdata .dummy_gid, .data;
	.last_dmg = .start_hp - .data[UMOB_HP];
	announce "DPS test ended. Total: " + .last_dmg + " / Avg DPS: " + (.last_dmg / 30), bc_all;
	killmonster "prontera", "DRO_DPS_DUMMY::OnDummyDeath";
	.dummy_gid = 0;
	end;

OnDummyDeath:
	// dummy died early (player out-DPS'd 2B HP — unlikely)
	if (.dummy_gid) {
		announce "Target killed before window ended. Result invalid.", bc_self;
		stopnpctimer;
		.dummy_gid = 0;
	}
	end;

OnInit:
	.last_dmg = 0;
	.dummy_gid = 0;
	end;
}
