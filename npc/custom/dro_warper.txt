prontera,159,192,4	script	Warper#dro	10007,1,1,{
	getmapxy(.@m$, .@nx, .@ny, BL_NPC, strnpcinfo(0));
	getmapxy(.@pm$, .@px, .@py, BL_PC);
	if (distance(.@px, .@py, .@nx, .@ny) <= 1) {
		callfunc "F_DroWarperMenu";
		end;
	}
	@warper_clicked = 1;
	unitwalk(getcharid(3), .@nx, .@ny);
	end;
OnTouch:
	if (@warper_clicked != 1) end;
	@warper_clicked = 0;
	callfunc "F_DroWarperMenu";
	unitstopwalk(getcharid(3), 8);
	end;
OnInit:
	setnpcwalkthrough 1;
	end;
}

// Effect 800 visual + spell-cast sound + 1s delay, then warp.
function	script	F_DroWarpCast	{
	specialeffect2 800;
	specialeffect2 EF_BEGINSPELL;
	sleep2 1000;
	warp getarg(0), getarg(1), getarg(2);
	end;
}
